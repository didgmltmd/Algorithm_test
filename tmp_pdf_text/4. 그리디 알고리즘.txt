최적 부분 구조
(Optimal Substructure)
"전체의 최적해는 부분의 최적해로 이루어진다."
• 문제 전체에 대한 최적 해결 방법이 하위 문제(Sub-
problem)들에 대해서도 최적 해결 방법이어야 함
• 즉, 문제의 스케일을 줄여나가도 본질적인 해의 구조
가 유지되어야 함

6. while ( change ≥ 1 )
change = change-1, n1++             // 1원 동전 수 1 증가
7. return (n500+n100+n50+n10+n1) // 총 동전 수 리턴

def min_coin_count(target_amount, coins):
"""
그리디 알고리즘을 이용한 최소 동전 개수 계산 함수
    :param target_amount: 거슬러 줘야 할 전체 금액
    :param coins: 사용 가능한 동전 종류 (리스트)
:return: 각 동전별 개수가 담긴 딕셔너리
    """
# 1. 동전을 큰 단위부터 처리하기 위해 내림차순 정렬
    coins.sort(reverse=True)
result = {}
remaining = target_amount
for coin in coins:
# 2. 현재 동전으로 거슬러 줄 수 있는 최대 개수 계산
        count = remaining // coin
if count > 0:
result[coin] = count
# 3. 남은 금액 업데이트
            remaining %= coin
return result, remaining

최소 신장 트리
➢ 주어진 그래프의 신장 트리를 찾는 방법
• 사이클이 없도록 모든 점을 연결시킴
➢ 그래프의 점의 수 = n
• 신장 트리에는 정확히 (n-1)개의 간선이 있음
• 트리에 간선을 하나 추가시키면, 반드시 사이클이 만들어짐

최소 신장 트리 알고리즘
➢ 크러스컬(Kruskal) 알고리즘
• 가중치가 가장 작은 간선이 사이클을 만들지 않을 때에만 ‘욕심내어’ 그 간선을 
추가시킴
➢ 프림(Prim) 알고리즘
• 임의의 점 하나를 선택한 후, (n-1)개의 간선을 하나씩 추가시켜 트리를 만듬
➢ 알고리즘의 입력은 1개의 연결 성분(connected component)으로 
된 가중치 그래프
4.2 최소 신장 트리
연결 성분 내의 점들 사이에 경로가 있는 극대부분 그래프(maximal subgraph)를 의미
그래프가 2개 이상의 연결 성분을 가지면, 서로 다른 연결성분의 점들 사이에는 경로가 없다

import heapq
def prim_basic(start_node, graph):
visited = set([start_node])
mst_edges = []
total_cost = 0
# 인접 간선들을 (가중치, 시작노드, 도착노드) 형태로 힙에 삽입
    edges = [(weight, start_node, to_node) for to_node, weight in graph[start_node]]
heapq.heapify(edges)
while edges:
weight, frm, to = heapq.heappop(edges)
# 아직 방문하지 않은 노드라면 트리에 추가 (그리디 접근)
if to not in visited:
visited.add(to)
mst_edges.append((frm, to, weight))
total_cost += weight
# 새롭게 연결된 노드의 인접 간선들을 힙에 추가
            for next_node, next_weight in graph[to]:
if next_node not in visited:
heapq.heappush(edges, (next_weight, to, next_node))
return total_cost, mst_edges

4.2 최소 신장 트리 # 3. 최대 신장 트리 (크루스칼 응용)
def max_spanning_tree(v, edges):
parent = [i for i in range(v + 1)]
def find(x):
if parent[x] != x: parent[x] = find(parent[x])
return parent[x]
def union(a, b):
a, b = find(a), find(b)
if a < b:
parent[b] = a
else:
parent[a] = b
# 최대 신장 트리를 위해 '내림차순' 정렬 (가장 큰 가중치부터 선택)
edges.sort(key=lambda x: x[2], reverse=True)
max_cost = 0
mst_edges = []
for a, b, cost in edges:
if find(a) != find(b):
union(a, b)
max_cost += cost
mst_edges.append((a, b, cost))
return max_cost, mst_edges

def dijkstra_pq(graph, start):
# 각 노드까지의 거리를 무한대로 초기화
    distances = {node: float('inf') for node in graph}
distances[start] = 0
# 우선순위 큐: (거리, 노드)
queue = [(0, start)]
while queue:
current_dist, current_node = heapq.heappop(queue)
# 이미 처리된 노드(더 짧은 경로가 발견된 경우)는 무시
        if distances[current_node] < current_dist:
continue
# 인접 노드 확인
        for neighbor, weight in graph[current_node].items():
distance = current_dist + weight
# 더 짧은 경로를 발견하면 거리 갱신 및 큐에 추가
            if distance < distances[neighbor]:
distances[neighbor] = distance
heapq.heappush(queue, (distance, neighbor))
return distances

def grid_dijkstra(grid, start, end):
rows, cols = len(grid), len(grid[0])
distances = [[float('inf')] * cols for _ in range(rows)]
distances[start[0]][start[1]] = 0
queue = [(0, start[0], start[1])]
# 이동 방향: 상, 하, 좌, 우
    directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
while queue:
dist, r, c = heapq.heappop(queue)
if (r, c) == end: return dist
if distances[r][c] < dist: continue
for dr, dc in directions:
nr, nc = r + dr, c + dc
# 맵 범위 내에 있고, 벽(1)이 아닌 경우만 이동 가능
            if 0 <= nr < rows and 0 <= nc < cols and grid[nr][nc] == 0:
cost = dist + 1  # 이동 비용은 1칸당 1로 가정
                if cost < distances[nr][nc]:
distances[nr][nc] = cost
heapq.heappush(queue, (cost, nr, nc))
return -1  # 도달 불가

def fractional_knapsack_basic(capacity, items):
# 1. 무게당 가치(단가)를 기준으로 내림차순 정렬 (가치 / 무게)
items.sort(key=lambda x: x[0] / x[1], reverse=True)
total_value = 0.0
for value, weight in items:
if capacity >= weight:
# 배낭 용량이 충분하면 물건을 통째로 넣음
            total_value += value
capacity -= weight
else:
# 배낭 용량이 부족하면 남은 용량만큼 물건을 쪼개서 넣음
            fraction = capacity / weight
total_value += value * fraction
break  # 배낭이 꽉 찼으므로 종료
    return total_value

def set_cover_basic(universe, subsets):
"""
:param universe: 커버해야 할 전체 원소들의 집합 (set)
:param subsets: 부분 집합들의 딕셔너리 {집합이름: set(원소들)}
"""
covered = set()  # 현재까지 커버된 원소들
    selected_subsets = []  # 선택된 부분 집합들
    # 전체 원소를 모두 커버할 때까지 반복
    while covered != universe:
best_subset_name = None
elements_covered_in_this_step = set()
for name, elements in subsets.items():
# 현재 부분 집합이 커버할 수 있는 '새로운' 원소들 (차집합 활용)
uncovered = elements - covered
# 가장 많은 새로운 원소를 커버하는 집합을 찾음
            if len(uncovered) > len(elements_covered_in_this_step):
best_subset_name = name
elements_covered_in_this_step = uncovered
# 더 이상 커버할 수 있는 집합이 없으면 무한루프 방지
        if best_subset_name is None:
break
selected_subsets.append(best_subset_name)
covered |= subsets[best_subset_name]  # 합집합 연산으로 커버된 영역 업데이트
    return selected_subsets

def hire_minimum_candidates(required_skills, candidates):
hired_candidates = []
fulfilled_skills = set()
while fulfilled_skills != required_skills:
best_candidate = None
skills_added_by_candidate = set()
for name, skills in candidates.items():
new_skills = skills - fulfilled_skills
if len(new_skills) > len(skills_added_by_candidate):
best_candidate = name
skills_added_by_candidate = new_skills
if best_candidate is None:
print("경고: 모든 요구 기술을 만족시킬 수 있는 지원자 조합이 없습니다.")
break
hired_candidates.append(best_candidate)
fulfilled_skills |= candidates[best_candidate]
return hired_candidates

def job_scheduling_basic(jobs):
# jobs = [(작업명, 마감일, 이익)]
# 1. 이익을 기준으로 내림차순 정렬
    jobs.sort(key=lambda x: x[2], reverse=True)
max_deadline = max(job[1] for job in jobs)
slots = [-1] * (max_deadline + 1)  # 1번 인덱스부터 사용
    total_profit = 0
scheduled_jobs = []
for job_id, deadline, profit in jobs:
# 마감일부터 1까지 역순으로 빈 슬롯 찾기
        for i in range(deadline, 0, -1):
if slots[i] == -1:
slots[i] = job_id
total_profit += profit
scheduled_jobs.append((job_id, i))  # (작업명, 배정된 시간)
break
return total_profit, scheduled_jobs

파이썬 구현 패턴 2 : 최소 힙(Min-Heap)을 활용한 스케줄링 최적화
4.6 작업 스케줄링
import heapq
def job_scheduling_heap(jobs):
# 1. 마감일을 기준으로 오름차순 정렬
    jobs.sort(key=lambda x: x[1])
min_heap = []
for job_id, deadline, profit in jobs:
heapq.heappush(min_heap, profit)
# 힙의 크기(현재까지 배정된 작업 수)가 마감일을 초과하면
        # 가장 이익이 적은 작업을 큐에서 제거
        if len(min_heap) > deadline:
heapq.heappop(min_heap)
# 힙에 남아있는 수익의 합이 최대 이익
    return sum(min_heap)

파이썬 구현 패턴 3 : 최소 지연 스케줄링 (EDF: Earliest Deadline First)
4.6 작업 스케줄링
def minimize_maximum_lateness(jobs):
# jobs = [(작업명, 소요시간, 마감일)]
# 마감일(Deadline)을 기준으로 오름차순 정렬 (EDF 전략)
jobs.sort(key=lambda x: x[2])
current_time = 0
max_lateness = 0
schedule_log = []
for job_id, duration, deadline in jobs:
start_time = current_time
current_time += duration  # 작업 소요 시간만큼 시간 흐름
        # 지연 시간 계산 (음수면 지연되지 않은 것이므로 0으로 처리)
lateness = max(0, current_time - deadline)
max_lateness = max(max_lateness, lateness)
schedule_log.append(f"{job_id}({start_time}~{current_time}) 지연:{lateness}일")
return max_lateness, schedule_log

최적의 그리디 증명: 확률이 높은 것에 가장 짧은 코드를 부여한다는 직관적이고 '탐욕적인' 아이디어가
수학적으로 손실 없는 가장 완벽한 압축임을 데이비드 허프만(D. Huffman)이 증명해 냄
제약 사항 (단점): 각 파일의 데이터 빈도수를 파악하기 위해 데이터를 두 번 읽어야(Two-pass) 하는
오버헤드가 발생할 수 있으므로, 실시간 환경에서는 동적(Adaptive) 허프만 코딩이 필요함

import heapq
from collections import Counter
# 허프만 트리를 구성할 노드 클래스
class Node:
def __init__(self, char, freq):
self.char = char
self.freq = freq
self.left = None
self.right = None
# 우선순위 큐(힙)에서 빈도수(freq)를 기준으로 정렬하기 위한 특수 메서드
    def __lt__(self, other):
return self.freq < other.freq

def build_huffman_tree(text):
# 1. 문자별 출현 빈도수 계산
    frequency = Counter(text)
# 2. 모든 문자를 단일 노드로 만들어 최소 힙에 삽입
    heap = [Node(char, freq) for char, freq in frequency.items()]
heapq.heapify(heap)
# 3. 트리에 노드가 하나만 남을 때까지 반복 (그리디 접근)
while len(heap) > 1:
# 빈도수가 가장 낮은 두 노드를 꺼냄
        left = heapq.heappop(heap)
right = heapq.heappop(heap)
# 두 노드를 자식으로 가지는 새 부모 노드 생성 (빈도수는 두 자식의 합)
merged = Node(None, left.freq + right.freq)
merged.left = left
merged.right = right
# 새 노드를 다시 힙에 삽입
        heapq.heappush(heap, merged)
return heap[0]  # 트리의 루트 노드 반환

def generate_codes(node, current_code="", codes={}):
if node is None:
return
# 단말 노드(문자가 있는 노드)에 도달하면 코드 저장
    if node.char is not None:
codes[node.char] = current_code
# 왼쪽으로 갈 때는 '0', 오른쪽으로 갈 때는 '1'을 붙임
    generate_codes(node.left, current_code + "0", codes)
generate_codes(node.right, current_code + "1", codes)
return codes
